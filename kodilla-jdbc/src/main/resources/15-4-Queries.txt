SELECT ISSUES.SUMMARY, ISSUESLISTS.NAME
FROM ISSUES
JOIN ISSUESLISTS ON ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;

SELECT ISSUES.SUMMARY, USERS.FIRSTNAME, USERS.LASTNAME
FROM ISSUES
JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(*) AS TASKS_QUANTITY
FROM USERS
JOIN ISSUES ON USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY USERS.LASTNAME, USERS.FIRSTNAME;
